<headline3 :title="title3" ref="headline3"></headline3>
<headline4 :title="title4" ref="headline4"></headline4>
<headline5 :title="title5" ref="headline5"></headline5>

const title1 = "网络数据能力";

const title5 = "公厕七日客流";
const title6 = "省分电路排名";
const headlineClass = 'headline-class1'

参数|说明|类型|可选值|默认值

title|标题|String|-|-
